# 在页面中使用

点击查看代码 
<template>
  <div class="form">
    <srm-form :data-set="dataSet" theme="fill" :columns="getColumns()" :buttons="['submit', 'update', 'delete']"/>
  </div>
</template>

<script>
import { DataSet } from "../../packages/data-set";
import srmListJSXExampleDS from "../datasource/SrmListJSXExampleDS";

export default {
  name: "SrmFormExample",
  data() {
    return {
      dataSet: new DataSet(srmListJSXExampleDS()),
    };
  },
  beforeMount() {
    this.dataSet.addPathParameter("tenantId", localStorage.getItem("tenantId"));
  },
  methods: {
    getHeadColumns() {
      return [
        { fieldCode: "rfxTitle" },
        { fieldCode: "rfxNum" },
        { fieldCode: "lastUpdateDate" },
        { fieldCode: "createdBy" },
        { fieldCode: "companyName" },
      ];
    },
    getColumns() {
      return [
        { fieldCode: "rfxTitle" },
        { fieldCode: "rfxNum" },
        { fieldCode: "lastUpdateDate" },
        { fieldCode: "createdBy" },
        { fieldCode: "companyName" },
        {
          fieldCode: "approvedDate",
          langCode: "ssrc.inquiryHall.model.inquiryHall.startTime",
          render: (val, data) => {
            return <SrmTextareaField v-model={data.rfxNum} name="rfxNum" placeholder="询价单号" size="small"/>;
          }
        }
      ];
    }
  }
};
</script>

<style scoped lang="stylus">
.form {
  height 100vh;
  overflow-y scroll;

}
</style>

# 组件属性

参数 说明 类型 是否必输 可选值 默认值 版本要求
customizeCode 个性化单元编码 String 否 - null --
customizeType 显示类型，表单/详情表单 String 否 list/detail list --
dataSet 列表数据源(详见DataSet) DataSet 是 - null --
columns 需要显示的行，详见Field Array 是 - [] --
buttons 需要显示的操作按钮 Array 否 - [] --
onSubmit 提交时回调函数 Function 否 - - v1.5.0
showDefaultDetail 是否显示弹窗 Boolean 否 - false -
isTemplate 是否开启单据样式定制 Boolean 否 - false 2.1.112-alpha/2.1.37
template 单据样式模板信息，业务模块模板接口返回数据，包含字段：{templateCode, templateVersion} Object isTemplate为true时必填 - - 2.1.112-alpha/2.1.37
pageCode 单据样式页面编码 String isTemplate为true时必填 - - 2.1.112-alpha/2.1.37
stageCode 单据样式阶段编码 String isTemplate为true时必填 - - 2.1.112-alpha/2.1.37
