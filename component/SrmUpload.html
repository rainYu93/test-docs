    # props

    参数 说明 类型 是否必输 可选值 默认值 版本要求
    descTitle 按钮表述 String 否 - - --
    color 字体颜色 String 否 - #333333 --
    type 主题类型(颜色分别为：蓝色，黑色，黄色，红色，绿色，紫色，棕色) String 否 primary/info/warn/danger/success/purple/brown primary -
    showIcon 是否显示回形针icon Boolean 否 - true --
    showCount 是否显示附件数量 Boolean 否 - true --
    title 附件弹窗标题 String 否 - 查看附件 --
    attachmentUuid 附件uuid/附件数据 String/Array 否 - - --
    fileArr 多类型附件数组 Array 否 - [] --
    fileList 附件列表，传值后组件不再调用接口获取列表 Array 否 - [] --
    bucketName 桶名 String 否 - private-bucket --
    queryParams 附件查询需要的额外参数 Object 否 - {} --
    canUpload 单类型是否可上传 Boolean 否 - false --
    canDelete 单类型是否可删除 Boolean 否 - false --
    isTab 是否是tab类型 Boolean 否 - false --
    readOnly 是否只读 Boolean 否 - true --
    onClose 附件弹窗关闭回调 Function 否 - - --
    moreInfoFlag 是否显示更多信息 Boolean 否 - false --
    canPreviewFlag 是否可预览 Boolean 否 - true --
    initCallback 组件初始化回调方法 Function 否 - - --

    fileArr

    参数 说明 类型 是否必输 可选值 默认值 版本要求
    title 标题 String 是 - - --
    attachmentUuid 附件uuid String 否 - - --
    readOnly 是否只读 Boolean 否 - - --

    # 全局方法 $showFileList(option)

    通过手动调用打开附件弹窗

    this.$showFileList({
   title: "查看附件"
   attachmentUuid: "3093ac2e6e4012306b9adc769204d5703c" 
});

this.$showFileList({
   title: "查看附件"
   isTab: true,
   fileArr: this.fileArr,
});
