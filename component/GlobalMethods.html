# 对话框

# 在页面中使用

点击查看代码 
    <template>
      <div class="box">
        <srm-button @click="showDialog">点击弹出对话框</srm-button>
        <srm-button @click="showDialogWithInput">点击有输入框的对话框</srm-button>
        <srm-button @click="showAlert">点击弹出Alert类型对话框</srm-button>
      </div>
    </template>
    
    <script>
    import SrmInputField from "@/components/SrmInputField";
    import SrmButton from "@/components/SrmButton";
    
    export default {
      name: "SrmPopupExample",
      components: {SrmButton},
      methods: {
        showDialog() {
          this.$dialog({
            content: "你好？",
            onOk: () => {
              console.log("onOk");
            },
            onCancel: () => {
              console.log("onCancel");
            }
          });
        },
        showDialogWithInput() {
          this.$dialog({
            content: <SrmInputField placeholder="请输入姓名"/>,
            onOk: () => {
              console.log("onOk");
            },
            onCancel: () => {
              console.log("onCancel");
            }
          });
        },
        showAlert() {
          this.$dialog({
            content: "你好!",
            type: "alert",
            onOk: () => {
              console.log("onOk");
            },
          });
        },
      }
    };
    </script>

# 参数说明

$lov(lovCode, callback, options)

参数 说明 类型 是否必输 可选值 默认值 版本要求
lovCode 值集代码 String 是 - - --
callback(val, valueField, displayField) 选择触发后回调函数,三个参数分别是选中的值，值字段名和显示字段名 Function 是 - - --
options 值集选项，详情见lovOptions Object 否 - {} --

lovOptions

支持说明

lovOptions字段在不同使用位置支持情况不同。

全局方法代表属性只能在$lov方法中使用； LovField代表只能在Field和SrmLovField中使用； 全局方法、LovField代表全局方法、Field和SrmLovField均可使用；

参数 支持 说明 类型 是否必输 可选值 默认值 版本要求
type 全局方法、LovField 值集代码 String 否 lov/select/tree(v1.1.7) lov --
queryParams 全局方法、LovField 默认查询参数 Object 否 - - --
multiple 全局方法、LovField 是否多选 Boolean 否 - false --
displayField 全局方法、LovField 显示字段 String 否 - 接口返回 --
subDisplayField（v2.2.1版本已废弃） 全局方法、LovField 括号里显示的字段 String 否 - 接口返回 --
valueField 全局方法、LovField 值字段名 String 否 - 接口返回 --
queryField（v1.1.2已废弃） -- 查询字段 String 否 - - --
title 全局方法、LovField 值集标题 String 否 - 接口返回 --
listData 全局方法、LovField lov的默认列表，lovCode为空时显示此数据，用来显示和选择非标准值集 Array<Object> 否 - null --
defaultValue 全局方法、LovField lov的默认已选列表 Array<Object> 否 - null --
childrenField 全局方法、LovField 树形结构的子集字段名 String 否 -- 接口返回/children v1.1.8
callback LovField 选择值集后的回调函数 Function(val, valueField, displayField, dataSet) 否 - - --
onClear LovField 清除选择结果的回调函数 Function(dataSet) 否 - - --
singleQueryFields 是否在切换快速查询条件时清空其他条件 Boolean - false v2.1.14
