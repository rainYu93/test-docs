    import { http, SrmUtil } from "srm-mobile-ui";
    
    /**
     * 查询寻源阶段
     * @param rfxHeaderId 询价单头id
     * @return {any}
     */
    export const getRfxProgressApi = (rfxHeaderId) => {
      const url = `/ssrc/v1/${SrmUtil.getTenantId()}/rfx/${rfxHeaderId}/progress/all`;
      return http.get(url);
    };
    
    /**
     * 查询历史操作记录
     * @param rfxHeaderId 询价单头id
     * @return {any}
     */
    export const getRfxHistoryActionApi = (rfxHeaderId) => {
      const url = `/ssrc/v1/${SrmUtil.getTenantId()}/rfx/${rfxHeaderId}/actions`;
      return http.get(url, {
        page: 0,
        size: 10000
      });
    };

# 3.2 views/ExampleRfxHallList.vue

点击查看代码 
    <template>
      <!--设置高度100vh，让srm-list可以计算自身高度。因为外层div高度是auto-->
      <div style="height: 100vh">
        <SrmList :data-set="dataSet"
                 :columns="getListColumns()"
                 :buttons="getListButtons()"
                 @itemClick="onListItemClick"/>
      </div>
    </template>
    
    <script>
    
    import { DataSet } from "srm-mobile-ui";
    import rfxHallListDS from "../datasource/rfxHallListDS";
    
    export default {
      modules: [ "ssrc.inquiryHall", "ssrc.queryRfq", "smbl.common" ],
      //name保持和文件名一致，见名知意
      name: "ExampleRfxHallList",
      data() {
        return {
          //初始化dataSet
          dataSet: new DataSet(rfxHallListDS()),
        };
      },
      //为了返回列表页时自动刷新，需要将当前页面加入keep-alive，并且在activated生命周期内刷新数据
      activated() {
        this.dataSet.refresh();
      },
      methods: {
        //列表操作按钮
        getListButtons() {
          return [
            //默认删除按钮和删除逻辑
            "delete",
            //默认保存按钮和保存逻辑
            "update",
            {
              text: this.$T("smbl.common.button.create").$D("新建"),
              type: "primary",
              visible: true,
              fill: true,
              //可以使用create方法在0位置创建一条空数据
              callback: () => this.dataSet.create({}, 0)
            }
          ];
        },
        //列表columns
        getListColumns() {
          return [
            {
              formRow: 0,
              fieldCode: "customHeader",
              fieldType: "EMPTY",
              render: (val, item) => <SrmListHeader>
                <span>{item.rfxNum}</span>
                <span slot="right">{item.rfxStatusMeaning}</span>
              </SrmListHeader>
            },
            {
              fieldCode: "rfxTitle",
              fieldType: "TEXT",
              //plain属性表示朴素的文字按钮
              render: (val, item) => <SrmButton plain onClick={() => this.toDetail(item)}>{val}</SrmButton>
            },
            { fieldCode: "companyName", },
            { fieldCode: "totalPrice", editor: this.dataSet.editFlag },
            //通过editor属性控制组件是否可编辑
            { fieldCode: "quotationStartDate", editor: this.dataSet.editFlag },
            { fieldCode: "quotationEndDate", },
            { fieldCode: "createByName", },
            {
              fieldCode: "customBottom",
              fieldType: "EMPTY",
              render: (val, item) => {
                return <SrmFlex justify="end">
                  <SrmButton fill onClick={() => this.toDetail(item)}>
                    {this.$T("ssrc.inquiryHall.view.message.title.checkPrice").$D("核价")}
                  </SrmButton>
                </SrmFlex>;
              }
            }
          ];
        },
        //列表详情弹窗columns
        getDetailColumns() {
          return [
            { fieldCode: "rfxNum", },
            { fieldCode: "rfxStatusMeaning", },
            { fieldCode: "rfxTitle", },
            { fieldCode: "companyName", },
            { fieldCode: "totalPrice", },
            //editor也可以是一个函数，val：当前字段值，item：当前数据
            { fieldCode: "quotationStartDate", editor: (val, item) => true },
            { fieldCode: "quotationEndDate", },
            { fieldCode: "createByName", },
          ];
        },
        //跳转详情
        toDetail(item) {
          this.$router.push({
            name: "ExampleRfxHallDetail",
            query: { rfxHeaderId: item.rfxHeaderId },
          });
        },
        //打开弹窗
        onListItemClick() {
          //可以使用$popup全局方法打开弹窗显示详情
          this.$popup({
            //不再显示全屏弹窗头部标题和返回按钮
            showHeader: false,
            showClose: false,
            position: "right",
            showBack: false,
            content: <SrmForm
              style={{ padding: "0 15px" }}
              customizeType="detail"
              dataSet={this.dataSet}
              columns={this.getDetailColumns()}
            />,
          });
        },
      },
    };
    </script>

# 3.3 views/ExampleRfxHallDetail.vue

点击查看代码 
<template>
  <div class="rfx-hall-detail">
    <!-- 头信息 -->
    <SrmDetailHead
      :columns="getHeaderColumn()"
      :data-source="headInfo">
      <div class="button-right-margin">
        <SrmButton size="small" icon="icon-paperclip" fill @click="showAttachment">
          <span>{{ $T('ssrc.inquiryHall.button.inquiryHall.file').$D('详情附件') }}</span>
        </SrmButton>
        <SrmButton size="small" icon="icon-clock" bordered @click="showHistory">
          <span>{{ $T('hzero.common.status.history').$D('历史记录') }}</span>
        </SrmButton>
      </div>
    </SrmDetailHead>
    <!-- 内容区域使用srm-scroll-tab定位轴 -->
    <srm-scroll-tab class="content">
      <!-- rfx寻源阶段，可以使用srm-step-progress组件 -->
      <srm-step-progress
        v-if="rfxProgress.length"
        justify="start"
        :steps="rfxProgress"/>
      <srm-scroll-tab-item :label="$T('ssrc.inquiryHall.view.inquiryHall.purOrganizationAndStaff').$D('采购组织及人员')">
        <srm-form :data-set="rfxDetailDataSet" :columns="getOrgPersonInfoColumns()"/>
      </srm-scroll-tab-item>
      <srm-scroll-tab-item :label="$T('ssrc.inquiryHall.view.inquiryHall.rfxItemLines').$D('询价标的物')">
        <srm-list :data-set="rfxItemListDataSet" :columns="getRfxItemColumns()" :scrollable="false"/>
      </srm-scroll-tab-item>
      <srm-scroll-tab-item :label="$T('ssrc.inquiryHall.view.inquiryHall.rfxDeamnd').$D('询价要求')">
        <srm-form :data-set="rfxDetailDataSet" :columns="getRfxDemandColumns()"/>
      </srm-scroll-tab-item>
    </srm-scroll-tab>
    <!-- 以弹窗形式显示历史记录 -->
    <HistoryRecord
      v-model="showHistoryFlag"
      :action-data="actionData"
      operator-field-name="realName"
      process-remark-field-name="processOperationMeaning"
      operation-meaning-field-name="processStatusMeaning"/>
  </div>
</template>

<script>

import { DataSet } from "srm-mobile-ui";
import {
  getRfxProgressApi,
  getRfxHistoryActionApi,
} from "../api/index";
import rfxDetailDS from "../datasource/rfxHallDetailDS";
import rfxItemListDS from "../datasource/rfxItemListDS";
import uuid from "uuid";
import HistoryRecord from "@/components/HistoryRecord";

export default {
  modules: [ "ssrc.inquiryHall", "smbl.common", "hipspfm.common", "hzero.common", "ssrc.queryRfq", "ssrc.bidHall", "smbl.rfx" ],
  name: "ExampleRfxHallDetail",
  components: { HistoryRecord, },
  data() {
    return {
      rfxDetailDataSet: new DataSet(rfxDetailDS()),
      rfxItemListDataSet: new DataSet(rfxItemListDS()),
      rfxHeaderId: null,
      headInfo: {},//头信息
      statusArr: [],//供应商阶段集合
      rfxProgress: [],//寻源价单阶段
      actionData: [],//历史记录
      showHistoryFlag: false, //历史记录弹窗
    };
  },
  computed: {
    fileArr() {
      return [ {
        title: this.$T("smbl.common.view.businessAttachment").$D("商务附件"),
        attachmentUuid: this.headInfo.businessAttachmentUuid || uuid(),
        readOnly: true
      }, {
        title: this.$T("smbl.common.view.technicalAttachment").$D("技术附件"),
        attachmentUuid: this.headInfo.techAttachmentUuid || uuid(),
        readOnly: true
      } ];
    },
  },
  beforeMount() {
    this.rfxHeaderId = this.$route.query.rfxHeaderId;
    this.rfxItemListDataSet.addQueryParameter("rfxHeaderId", this.rfxHeaderId);
    this.rfxDetailDataSet.addPathParameter("rfxHeaderId", this.rfxHeaderId);
    this.rfxDetailDataSet.events.afterQuery = (ds, data) => {
      //ds接口调用完成后的数据处理逻辑
      if (data && data.length) {
        this.headInfo = data[0];
        this.statusArr = (data[0].allowSourceSupplierStages || "").split(",");
      }
    };
  },
  async mounted() {
    try {
      this.$loading.show();
      await this.getRfxProgressInfo();
      await this.getRfxHistoryActionInfo();
    } finally {
      this.$loading.hide();
    }
  },
  methods: {
    // 询价详情头
    getHeaderColumn() {
      return [
        {
          fieldName: this.$T("ssrc.queryRfq.model.queryRfq.inquiryTitle").$D("询价单标题"),
          fieldType: "TEXT", // 默认TEXT,可选 TEXT/TAG
          fieldCode: "rfxTitle",
        },
        { fieldCode: "rfxNum", },
        {
          fieldName: this.$T("ssrc.inquiryHall.modal.inquiryHall.sourcingTemplate").$D("寻源模板"),
          fieldCode: "templateName",
        }
      ];
    },
    // 询价标的物
    getRfxItemColumns() {
      return [
        {
          fieldCode: "itemName",
          fieldType: "EMPTY",
          render: (val) => <SrmListHeader showOpenTag>{val}</SrmListHeader>
        },
        { fieldCode: "itemCode", },
        { fieldCode: "rfxQuantity", },
        { fieldCode: "uomName", },
        { fieldCode: "rfxLineItemNum", },
      ];
    },
    // 询价要求
    getRfxDemandColumns() {
      return [
        {
          formRow: 0,
          fieldType: "EMPTY",
          render: () => {
            return <SrmListHeader showOpenTag>
              {this.$T("ssrc.inquiryHall.model.inquiryHall.sourcingCategory").$D("寻源类别")}：
              {this.headInfo.sourceCategoryMeaning}
            </SrmListHeader>;
          }
        },
        { fieldCode: "quotationTypeMeaning", },
        { fieldCode: "preQualificationFlag", },
        { fieldCode: "expertScoreFlag", },
        { fieldCode: "quotationStartDate", },
        { fieldCode: "quotationEndDate", },
      ];
    },
    // 采购组织人员
    getOrgPersonInfoColumns() {
      return [
        {
          fieldCode: "customizeHeader",
          fieldType: "EMPTY",
          render: (val, item) => {
            return <SrmListHeader showOpenTag={false}>{item.rfxNum}</SrmListHeader>;
          }
        },
        { fieldCode: "companyName", },
        { fieldCode: "unitName", },
        { fieldCode: "purOrganizationName", },
        { fieldCode: "purchaserName", },
        { fieldCode: "purName", },
        { fieldCode: "purPhone", },
        { fieldCode: "purEmail", },
        { fieldCode: "checkedByName", },
        { fieldCode: "rfxByName", },
        { fieldCode: "checkedByName", },
      ];
    },
    //获取寻源价单阶段信息
    async getRfxProgressInfo() {
      await getRfxProgressApi(this.rfxHeaderId).then((res) => {
        if (res && !res.failed) {
          this.rfxProgress = res.map((e) => ({
            nodeSeq: e.nodeSeq,
            nodeDes: e.nodeStatusMeaning,
            nodeState: e.nodeFlag
          }));
        }
      });
    },
    //获取操作记录信息
    async getRfxHistoryActionInfo() {
      await getRfxHistoryActionApi(this.rfxHeaderId).then((res) => {
        if (res && res.content) {
          this.actionData = res.content;
        }
      });
    },
    //历史记录显示
    showHistory() {
      this.showHistoryFlag = true;
    },
    //附件显示
    showAttachment() {
      //使用全局方法$showFileList预览多tab附件
      this.$showFileList({
        isTab: true,
        fileArr: this.fileArr,
      });
    },
  },
};
</script>

    import { SrmUtil, $T } from "srm-mobile-ui";
    
    function rfxHallListDS() {
      return {
        autoQuery: true, //是否在组件挂载自动查询
        selection: Infinity,
        pageSize: 10,
        editable: true,
        enableQuery: true,
        //默认路径参数
        pathParameter: {
          //使用工具类获取当前租户id
          tenantId: SrmUtil.getTenantId(),
        },
        //dataSet字段
        fields: [
          {
            //可以省略langCode字段，直接使用$T方法
            fieldName: $T("ssrc.queryRfq.model.queryRfq.inquiryTitle").$D("询价单标题"),
            fieldCode: "rfxTitle",
            fieldType: "INPUT",
          },
          {
            fieldName: $T("ssrc.queryRfq.model.queryRfq.company").$D("公司"),
            fieldCode: "companyName",
            fieldType: "TEXT",
          },
          {
            fieldName: $T("ssrc.queryRfq.model.queryRfq.rfxNum").$D("询价单号"),
            fieldCode: "rfxNum",
            fieldType: "TEXT",
          },
          {
            fieldName: $T("ssrc.queryRfq.model.queryRfq.rfxStatusMeaning").$D("询价单状态"),
            fieldCode: "rfxStatusMeaning",
            fieldType: "TEXT",
          },
          {
            fieldName: $T("ssrc.queryRfq.model.queryRfq.startTime").$D("开始时间"),
            fieldCode: "quotationStartDate",
            fieldType: "DATE_PICKER",
            editor: false,
          },
          {
            fieldName: $T("ssrc.queryRfq.model.queryRfq.quotationDeadline").$D("报价截止时间"),
            fieldCode: "quotationEndDate",
          },
          {
            fieldName: $T("ssrc.queryRfq.model.queryRfq.creater").$D("创建人"),
            fieldCode: "createByName",
          },
          {
            fieldName: $T("ssrc.queryRfq.model.queryRfq.totalPrice").$D("总价"),
            fieldCode: "totalPrice",
            fieldType: "INPUT_NUMBER",
            //通过precision指定数字的格式化精度，precision可以是数字或函数
            precision: () => 4,
          },
        ],
        //快速查询字段，可以是数组表示多个快速查询
        fastQueryField: {
          fieldName: $T("ssrc.queryRfq.model.queryRfq.inquiryTitle").$D("询价单标题"),
          fieldCode: "rfxTitle",
          fieldType: "TEXT",
        },
        //查询字段
        queryFields: [
          {
            fieldName: $T("ssrc.queryRfq.model.queryRfq.creater").$D("创建人"),
            fieldCode: "createdBy",
            fieldType: "LOV",
            lovCode: "HIAM.TENANT.USER",
            options: {
              title: $T("ssrc.queryRfq.model.queryRfq.selectCreater").$D("选择创建人"),
              multiple: false,
              displayField: "realName",
              subDisplayField: "loginName",
              valueField: "id",
              queryField: "realName",
              queryParams: {
                organizationId: SrmUtil.getTenantId(),
              },
            },
          },
          {
            fieldCode: "rfxStatusSet",
            fieldType: "SELECT",
            langCode: "ssrc.queryRfq.model.queryRfq.status",
            fieldName: "状态",
            lovCode: "SSRC.RFX_STATUS",
            options: {
              multiple: false,
            }
          },
        ],
        //分别定义ds增删查改操作的接口
        transport: {
          read: {
            url: "/ssrc/v1/{tenantId}/rfx/list",
            method: "get",
          },
          submit: {
            url: "/ssrc/v1/{tenantId}/rfx/list",
            method: "post",
          },
          update: {
            url: "/ssrc/v1/{tenantId}/rfx/list",
            method: "put",
          },
          destroy: {
            url: "/ssrc/v1/{tenantId}/rfx/list",
            method: "delete",
          },
        },
        events: {
          beforeQuery: (dataSet, queryParameter, pathParameter, bodyParameter) => {
            //ds查询前调用，可以在这里修改查询参数
            console.log(dataSet, queryParameter, pathParameter, bodyParameter);
          },
          afterQuery: (dataSet, result) => {
            //ds查询前调用，可以在这里修改返回结果
            console.log(dataSet, result);
          },
        }
      };
    }
    
    export default rfxHallListDS;
